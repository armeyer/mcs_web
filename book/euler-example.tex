\begin{problem}

\begin{align*}
n = 28 & = 2^2 \cdot 7,\\
\phi(n) & = (2^2 -2^1)(7-1) = 12,\\
\relpr{28} & = \set{1, 3, 5, 9, 11, 13, 15, 17, 19, 23, 25, 27}\\
\ordmod{9}{28} & = 3\\
9^1 & = 9, 9^2 = 25, 9^3 = 1 \pring{28}
\end{align*}


\begin{align*}
P & \eqdef \set{9, 25, 1}\\
1P & = P\\
3P & = \set{27, 19, 3}\\
5P & = \set{17, 13, 5}\\
9P & = P\\
11P & = \set{15, 23, 11}\\
13P & = 5P\\
15P & = 11P\\
17P & = 5P\\
19P & = 3P\\
23P & = 11P\\
25P & = 1P\\
27P & = 3P.
\end{align*}

So
\begin{equation}\label{cosets28}
\relpr{28} = 1P \union 3P \union 5P \union 11P.
\end{equation}


\end{problem}
