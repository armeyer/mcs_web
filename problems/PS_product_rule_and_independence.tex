\begin{problem}
Describe events $A$, $B$, and $C$ that:

\begin{itemize}
\item satisfy the ``\idx{product rule},'' namely,
\[
\pr{A \cap B \cap C} = \pr{A} \cdot \pr{B} \cdot \pr{C},
\]

\item no two out of the three events are independent.
\iffalse
 satisfy the product rule, namely,
\begin{align*}
\pr{A \intersect B} \neq \pr{A}\cdot \pr{B},\\
\pr{A \intersect C} \neq \pr{A}\cdot \pr{C},\\
\pr{B \intersect C} \neq \pr{B}\cdot \pr{C}.
\end{align*}
\fi

\end{itemize}

\hint Choose $A,B,C$ events over the uniform probability space on
$\Zintv{1}{6}$.

\begin{solution}
\[
 A  \eqdef \set{1,2,3},  B \eqdef \set{3,4,5},  C \eqdef \set{3,4,5,6}.
\]

So
\begin{align*}
\lefteqn{\pr{A \intersect B \intersect C}}\\
   & = \pr{3} = \frac{1}{6} = \frac{1}{2} \cdot  \frac{1}{2} \cdot \frac{2}{3}\\
   & = \pr{A} \cdot \pr{B} \cdot \pr{C},\\
\pr{A \intersect B}
    & = \pr{3} = \frac{1}{6}\\
    & \neq \frac{1}{2} \cdot \frac{1}{2} = \pr{A} \cdot \pr{B},\\
\pr{A \intersect C}
    & = \pr{3} = \frac{1}{6}\\
    & \neq \frac{1}{2} \cdot \frac{2}{3} = \pr{A} \cdot \pr{C},\\
\pr{B \intersect C}
    & = \pr{B} = \frac{1}{2}\\
    & \neq \frac{1}{2} \cdot \frac{2}{3} = \pr{B} \cdot \pr{C}.
\end{align*}
\end{solution}

\end{problem}
