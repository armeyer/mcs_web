\begin{problem}

\iffalse
Suppose $p$ is a prime and $p \geq 5$.  This problem will show that
$p^2-1$ is divisible by $24$.
\fi

\bparts

\ppart\label{ppm13} Prove if $n$ is not divisible by 3, then $n^2 \equiv 1 \pmod 3$.

\begin{solution}
Since $n \not\equiv 0 \pmod 3$, it follows that $n \equiv 1\text{ or
}2 \pmod 3$.  But $2^2 \equiv 1 \pmod 3$, so $n^2 \equiv 1 \pmod 3$ in
any case.
\end{solution}

\iffalse

\ppart Conclude that $n^2-1$ is divisible by 3.

\begin{solution}
It follows immediately from part~\eqref{ppm13} that
\[
n^2-1 \equiv 0 \pmod 3,
\]
which means that $n^2-1$ is multiple of 3.
\end{solution}
\fi

\ppart\label{noddn2e1} Show that if $n$ is odd, then $n^2 \equiv 1 \pmod 8$.

\begin{solution}
If $n$ is odd, its remainder on division by 8 is odd, and so $n \equiv
1,3,5, \text{ or } 7 \pmod 8$.  But it's easy to verify that $1^2,
3^2, 5^2$, and $7^2$ are all $\equiv 1 \pmod 8$.
\end{solution}

\ppart Conclude that if $p$ is a prime greater than 3, then $p^2-1$ is
divisible by $24$.

\begin{solution}
If $p > 3$ is prime, it is not divisible by 3, so by
part~\eqref{ppm13}, $p^2 \equiv 1 \pmod 3$, which is equivalent to the
fact that 3 divides $p^2-1$.

Also, if $p > 3$ is prime, it is certainly odd, so by
part~\eqref{noddn2e1}, $p^2 \equiv 1 \pmod 8$, which is equivalent to
the fact that 8 divides $p^2-1$.

But since $8$ and $3$ are relatively prime, anything divisible by both
of them is also divisible by their product, $24$.  So $24 \divides p^2
- 1$.
\end{solution}

\eparts

\end{problem}

\iffalse
\eparts

Since $p$ is not divisible by 2, it can be expressed as $2n+1$
for some integer $n >1$.

\bparts

\ppart Show that if $n$ is even, then $p^2-1$ is divisible by 8.

\begin{solution}
\begin{equation}\label{p214n2}
p^2-1 = 4n^2+4n.
\end{equation}
Since $n$ is even, the right hand side of~\eqref{p214n2} is
obviously divisible by $8$.
\end{solution}

\ppart Show that also if $n$ is odd, then $p^2-1$ is divisible by 8.

\begin{solution}
Now $n=2m+1$ for some $m \in \integers^+$, so the right side
of~\eqref{p214n2} equals
\[
4(2m+1)^2 +4(2m+1) = 4(4m^2 + 4m + 1) + 8m + 4 = 16m^2+24m+8
\]
which is divisible by $8$.
\end{solution}
