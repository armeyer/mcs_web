\begin{problem}
Prove by induction:
\begin{equation}
  \sum_{i=0}^{n} i^3  = \paren{\frac{n(n+1)}{2}}^{2}, \forall n \geq 0.
\end{equation}
using the equation itself as the induction hypothesis, $P(n)$.

\bparts

\ppart Prove the \inductioncase{base case} $(n=0)$.

\begin{solution}
  \[
  \sum_{i=0}^{0} i^3 = 0 \paren{\frac{(0)(1)}{2}}^2.
  \]
Therefore, $P(0)$ is true.
\end{solution}

\examspace[2in]

\ppart Now prove the \inductioncase{inductive step}.

\begin{solution}
\begin{align*}
\sum_{i=0}^{n+1} i^3
 & = \sum_{i=0}^{n} i^3 + (n+1)^3\\
 & = \paren{\frac{n(n+1)}{2}}^{2} + (n+1)^3 & \text{(by induction hypothesis $P(n)$)}\\
 & = (n+1)^2 \paren{\paren{\frac{n}{2}}^{2} + (n+1)}\\
 & = (n+1)^2 \paren{\frac{n^2}{4} + \frac{4n+4}{4}}\\
 & = (n+1)^2 \frac{n^2 + 4n + 4}{4}\\
 & = (n+1)^2 \frac{(n+2)^2}{4}\\
 & = \paren{\frac{(n+1)(n+2)}{2}}^{2}.
\end{align*}
Thus we have proved $P(n+1)$, completing the induction step.

We conclude by induction that $P(n)$ holds for all $n \in \naturals$.
\end{solution}

\eparts

\end{problem}
