\begin{problem}
\newcommand{\bea}{\begin{eqnarray*}}
\newcommand{\eea}{\end{eqnarray*}}

\begin{theorem*}
The image of $[\alpha, \beta] \subseteq [0,1]$ under $f(x)= 4x(1-x)$
is bounded above by $4 \delta$, where $\delta$ is the length of the
interval, $[\alpha, \beta]$.
\end{theorem*}

Complete the following proof of the theorem above. 

\emph{Proof by cases.}  Let $X = [\alpha, \beta]$, and let $Y$ denote
the image of $X$ under $f$.

Case~1: $\alpha < \beta \leq \frac{1}{2}$, or $\frac{1}{2} \leq \alpha
<\beta$.  Since $f$ is monotonically increasing or decreasing over
$X$; $Y = [f(\alpha), f(\beta)]$, or $Y = [f(\beta), f(\alpha)]$; and
\bea \left|\frac{\Delta Y}{\Delta X}\right| =
\left|\frac{f(\alpha)-f(\beta)}{\alpha - \beta}\right| \leq
\max_{x\in[\alpha,\beta]} \left|\frac{df(x)}{dx}\right| \leq
\max_{x\in[0,1]} \left|\frac{df(x)}{dx}\right| = 4.  \eea

Case~2: \exambox{2in}{0.5in}{-0.35in} 

\begin{solution}
$\alpha < \frac{1}{2} < \beta$.  
\end{solution}

\begin{center} \exambox{\textwidth}{3.5in}{-0.35in}\end{center}

\begin{solution}
The maximum value in the image is $1$ because $\frac{1}{2} \in
[\alpha, \beta]$, and $f(\frac{1}{2}) = 1$; and the minimum value in
the image is the minimum of $f(\alpha)$ and $f(\beta)$ because $f$ is
monotonically increasing over the interval, $[\alpha, \frac{1}{2}]$
and monotonically decreasing over the interval, $[\frac{1}{2},\beta]$.
Therefore, the image, $Y = [\min(f(\alpha), f(\beta)), 1]$.  Without
loss of generality, let $f(\alpha) = \min(f(\alpha), f(\beta))$. Then,
the image of $X' = [\alpha, \frac{1}{2}]$ under $f$ is precisely
$Y$. Therefore, $\left|\frac{Y}{X}\right| < \left|\frac{Y}{X'}\right|
\leq 4$.
\end{solution}

\end{problem}
