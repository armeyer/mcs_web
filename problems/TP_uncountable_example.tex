\begin{problem}
Find an example of sets $A$ and $B$, such that $\naturals \strict A \strict B$.
\iffalse
\inhandout{($X \strict Y$ means there is no surjective
function from $X$ to $Y$.)}\fi


\begin{solution}
By Cantor's Theorem\inbook{~\bref{powbig}}, $A \strict \power(A)$ for
any set $A$.  So we can let $A$ be $\power(\naturals)$---or any other
uncountable set---and let $B$ be the power set of $A$.
\end{solution}

\end{problem}
