\begin{problem}
Find the coefficients of 

\bparts

\ppart
 $x^8$ in $(x+x^3)^{4}$

\begin{solution}
$(x+x^3)^{4}$ can be expressed as $x^4(1+x^2)^4$.
\[
\binom{4}{2} = 6
\]
\end{solution}

\ppart $x^4y^{12}$ in $(7x+14y)^{16}$

\begin{solution}
\[
\binom{16}{4}7^414^{12}.
\]

When $(7x+14y)^{16}$ is expressed as a sum of powers of the summands
$7x$
and $14y$, the coefficient of $(7x)^4(14y)^{12}$ is $\binom{16}{4}$,
so the
coefficient of $x^4y^{12}$ is this binomial coefficient times
$7^4\cdot 14^{12}$.
\end{solution}

\eparts
\end{problem}
