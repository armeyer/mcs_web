\begin{problem}
Prove the Distributive Law of intersection over the union of $n$ sets
by induction:
\begin{equation}\label{AIBn}
  A \intersect \lgunion_{i=1}^n B_i
  = \lgunion_{i=1}^n (A \intersect B_i).
\end{equation}

\hint Theorem~\bref{thm:intersect-distributivity} gives the $n=2$
case\inbook{.}
\inhandout{
\begin{equation}\label{distr}
A \intersect (B_1 \union B_2) = (A \intersect B_1) \union (A \intersect  B_2).
\end{equation}
