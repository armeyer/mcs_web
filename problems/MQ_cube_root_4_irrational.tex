\begin{problem}\label{generprob}
   Prove that $\sqrt[3]{4}$ is irrational.

\begin{solution}
\begin{proof}
  Assume for the sake of contradiction that $\sqrt[3]{4}$ is rational.
  Under this assumption, there exist integers $a$ and $b$ such that
\[
\sqrt[3]{4} =  a/b,
\]
where $a$ and $b$ have no common factor.  Now we prove that $a$ and $b$
are both even, that is, they have 2 as a common factor, a contradiction.

\begin{align*}
\sqrt[3]{4} & = \frac{a}{b}, & \text{(by assumption)}\\
4           & = \frac{a^{3}}{b^{3}}, & \text{(cubing both sides)}\\
4b^{3}      & = a^{3}.
\end{align*}
The lefthand side of the last equation is even, so $a^{3}$ is even, which
implies that $a$ is even.

In particular, $a = 2c$ for some integer $c$.  Thus,
\begin{align*}
4b^{3} & = (2c)^3 = 8c^{3},\\
b^{3}  & = 2c^{3} & \text{(dividing by 4)}.
\end{align*}
The righthand side of the last equation is even,
so $b^{3}$ is even, which implies that $b$  is even.
\end{proof}

\end{solution}

\end{problem}
