\begin{problem}
For each of the following propositions:
\begin{enumerate}

\item $\forall x\,  \exists y. \; 2x-y = 0$

\item $\forall x\, \exists y. \; x-2y = 0$

\item
$\forall x.\, x < 10 \QIMPLIES (\forall y. \; y < x \QIMPLIES y < 9)$

\item
$\forall x\, \exists y.\; [y > x \And \exists z. \; y+z = 100]$

\end{enumerate}
determine which propositions are true when the variables range over:

\bparts

\ppart
the nonnegative integers. 

\begin{solution}
1, 3.

Counterexample for 2:  let $x = 1$.

Counterexample for 4:  let $x = 100$.
\end{solution}

\ppart
the integers. 

\begin{solution}
1,
3,
4.

Counterexample for 2:  let $x = 1$.
\end{solution}

\ppart
the real numbers. 

\begin{solution}
1,
2,
4.

Counterexample for 3:  let $x = 9.5$.
\end{solution}

\eparts

\end{problem}
