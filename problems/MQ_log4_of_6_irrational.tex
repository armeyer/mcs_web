\begin{problem}
 Prove that $\log_4 6$ is irrational.

\begin{solution}
\begin{proof}
  Suppose to the contrary that $\log_4 6$ is rational, so
  \[
  \textcolor{red}{\log_4 6 = m/n}
  \]
  for some positive integers $m$ and $n$.  So
  \[
  n\log_4 6 = m.
  \]
  Now raising 4 to each side of this equation gives
\begin{align}
  4^{n \log_4 6} & = 4^m, \notag\\
(4^{\log_4 6})^n & = 4^m, \notag\\
           6^n & = 4^m \label{4m4n6}
\end{align}
But the left hand side of~\eqref{4m4n6} is divisible by 3 (since
$n>0$), and the right hand side is not.

This contradiction implies that $\log_4 6$ must be irrational.
\end{proof}
\end{solution}
\end{problem}
