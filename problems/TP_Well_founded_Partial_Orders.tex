\begin{problem}

Which of the following are well founded partial orders?
     
\bparts
    
\ppart $\naturals \times \naturals^+$ under coordinatewise partial
order.

\begin{solution}
Yes.  

This is well founded, since each of the sets in the product is a
well founded partial (actually linear) order, and the product of
well founded orders is well founded.
\end{solution}
    
\ppart
$\naturals \times \integers^-$ under lexicographic order

\begin{solution}
No.

{(0,-1),(0,-2),(0,-3),\dots } is a nonempty subset with no least element.
\end{solution}
 
\ppart $[0,1,\dots,j] \times [-1,-2,\dots, -k]$ under coordinatewise
partial order, where $j,k \in \naturals$

\begin{solution}
Yes.

Every finite partially ordered set is well founded, so the product of
two finite sets is as well.
\end{solution}

\ppart a partially ordered set with no infinite decreasing sequence of
elements
\begin{solution}
Yes.

Having no infinite decreasing sequence is equivalent to being
     well founded.
\end{solution}
    
\ppart
 real numbers in the interval [1,2]
\begin{solution}
No.

The numbers $> 1$ in this interval are a subset with no least element.
\end{solution}
    
\ppart
 a partially ordered set with no infinite increasing
    sequence of elements
\begin{solution}
No.

The negative integers have no infinite increasing sequence, but are
      not well founded.
\end{solution}
    
\ppart $\strings{A}$ ---where $\strings{A}$ is the set of all finite
strings of letters of the Roman alphabet, $A$, ordered as they would
be in a dictionary (this is not the same as lexicographic order)

\begin{solution}
No.

$\set{b, ab, aab, aaab,\dots}$ is a set of strings with no element that would
    be listed first in dictionary order.
\end{solution}
    
\ppart
 $\strings{A^}$ ordered in the reverse of dictionary order
\begin{solution}
No.

$\set{a, aa, aaa, aaaa,\dots}$ is a set of strings with no element that would
     be listed last in dictionary order.
\end{solution}

\eparts

\end{problem}

\begin{problem}
Let $D$ be the usual \emph{dictionary order} on finite sequences of letters
of the alphabet.  Show that neither $D$ nor $D^{-1}$ is well founded.

\solution{\[
a\ D\ aa\ D\ aaa D\dots
\]
is an infinite $D$-increasing sequence, and so is an infinite
$D^{-1}$-decreasing sequence, proving that $D^{-1}$ is not well founded.
To show $D$ is not well founded, we describe 
an infinite $D^{-1}$-increasing sequence:
\[
b\ D^{-1}\ ab\ D^{-1}\ aab\  D^{-1}\ aaab \dots.
\]
}
\end{problem}
