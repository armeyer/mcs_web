\begin{problem}

T-Pain is planning an epic boat trip and he needs to decide what to
bring with him.

\begin{itemize}

\item He can bring burgers, but they only come in packs of 6.

\item He and his two friends can't decide whether they want to dress formally or
casually. He'll either bring 0 pairs of flip flops or 3 pairs.

\item He doesn't have very much room in his suitcase for towels, so he can
  bring at most 2.

\item In order for the boat trip to be truly epic, he has to bring at least 1
nautical-themed pashmina afghan.

\end{itemize}

\bparts

\ppart Let $B(x)$ be the generating function for the number of ways to
bring $n$ burgers, $F(x)$ for the number of ways to bring $n$ pairs of
flip flops, $T(x)$ for towels, and $A(x)$ for Afghans.  Write simple
formulas for each of these.

\begin{center}
$B(x):$\exambox{1.5in}{0.6in}{-0.5in} \qquad\qquad$F(x):$\exambox{1.5in}{0.6in}{-0.5in}
\end{center}
\begin{center}
$T(x):$\exambox{1.5in}{0.6in}{-0.5in} \qquad\qquad$A(x):$\exambox{1.5in}{0.6in}{-0.5in}
\end{center}

\ppart[3] Let $g_n$ be the the number of different ways for T-Pain to bring
$n$ items (burgers, pairs of flip flops, towels, and/or afghans) on his
boat trip.   Let
$G(x)$ be the generating function
$\sum_{n=0}^{\infty} g_nx^n$.  Verify that
\[
G(x) = \frac{x}{\paren{1-x}^2}.
\]

\iffalse
\begin{center}
\exambox{3in}{0.6in}{-0.5in}
\end{center}
\fi

\examspace[2in]

\begin{solution}
The successive generating functions for beer, flip flops, towels, and
afghans, are
\[
\frac{1}{1-x^6}, \quad (1+x^3), \quad (1+x+x^2), \quad \frac{x}{1-x}.
\]
So $G(x)$ is their product:
\begin{align*}
\frac{1}{1-x^6}(1+x^3)(1+x+x^2) \frac{x}{1-x}
  & = \frac{(1+x^3)(1+x+x^2)x} {(1-x^3)(1+x^3)(1-x)}\\
  & = \frac{(1+x+x^2)x} {(1-x)(1+x+x^2)(1-x)}\\
  & = \frac{x}{(1-x)^2}
\end{align*}
\end{solution}

\ppart Find a simple formula for $g_n$.

\begin{center}
\exambox{0.75in}{0.6in}{-0.5in}
\end{center}

\examspace[2in]

\begin{solution}
$n$.

Let
\[
G(x) \eqdef \frac{1}{(1-x)^2},
\]
so the generating function for T-Pain is $xG(x)$.  We know that the
coefficient of $x^n$ in the series for ${(1-x)^2}$ is, by the Convolution
Rule, the number of ways to select $n$ items of two different kinds,
namely, $\binom{n+1}{1}=n+1$, so we conclude that the $n$th coefficient in
the series for $G(x)$ is $n+1$.

So the $n$th coefficient in the series for the generating function
$xG(x)$ is zero for $n =0$, and, for $n \ge 1$, is
the $(n-1)$th coefficient of $G$, namely,
\[
(n-1)+1 = n.
\]
\end{solution}

\eparts

\end{problem}
