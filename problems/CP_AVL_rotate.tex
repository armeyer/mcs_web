\begin{problem}
Suppose $B$ is an AVL tree of depth $d$, $S$ is an AVL tree of depth
$d+2$, and $N$ is a search tree with branches $B$ and $S$ as shown in
Figure~\ref{rotate3}.  Also, $S$ has as branches an AVL tree $R$ of
depth $d+1$ and an AVL tree $U$ with depth $d$, as showm.

\begin{figure}
\begin{verbatim}
                             N
                            / \
                           /   \
                (depth d) B     S  (depth d+2)
                               /\
                              /  \
              (depth d+1)    R    U (depth d)
                            / \
                           V   W
\end{verbatim}

\caption{The Tree N before rotation.}

\label{rotate3}

\end{figure}

\begin{figure}

\begin{verbatim}

                             R'
                            / \
                           /   \
                          /     \
           (depth d+1)   N'      S'  (depth d+1)
                        / \     / \
                       /   \   /   \
                      B     V  W    U
\end{verbatim}

\caption{R' is the rotation of tree N in Figure~\ref{rotate3}.}

\label{rotate4}
                                                                   
\end{figure}                                                       

We define $R', N', S'$ to be new trees with
\begin{align*}
\nlbl{(R')} & = \nlbl{(R)}\\
\nlbl{(N')} & = \nlbl{(N)}\\
\nlbl{(S')} & = \nlbl{(S)},
\end{align*}
with branches as indicated in Figure~\ref{rotate4}.

Verify that $R'$ is an AVL tree and $\nlbls{(R')} = \nlbls{(N)}$.
