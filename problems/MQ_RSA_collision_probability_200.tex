\begin{problem}
Suppose the RSA modulus $n=pq$ is the product of distinct $200$ digit
primes $p$ and $q$.  A message $m\in [1,n)$ is called \emph{dangerous} if
  $\gcd(m,n)=p$, because such an $m$ can be used to factor $n$ and so
  crack RSA.  Circle the best estimate of the fraction of messages in
  $[1,n)$ that are dangerous.

\[\frac{1}{200} \qquad \frac{1}{400} 
\qquad \frac{1}{200^{10}}
\qquad \frac{1}{10^{200}}
\qquad \frac{1}{400^{10}}
\qquad \frac{1}{10^{400}}
\]

\begin{solution}
\[
\frac{1}{10^{200}}
\]

For $m \in [1,pq)$,
\[
\gcd(m,pq)=p \qiff p \divides m.
\]

The fraction of numbers divisible by $p$ in an interval is $1/p$, 
and since $10^{199} \le p < 10^{200}$, the best estimate of $1/p$ is
$1/10^{200}$.
\end{solution}

\end{problem}
