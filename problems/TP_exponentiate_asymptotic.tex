\begin{problem}
Let $f,g: \integers^+ \to \integers^+ $ be functions on the
positive integers.

\bparts 

\ppart Give an example of $f,g$ such that
\[
f \sim g\ \QAND\ 2^f = o(2^g).
\]

\examspace[2.0in]

\begin{solution}
$f(n) \eqdef n^2$, $g(n) \eqdef n^2 + n$, is one example.

$f(n) \eqdef n$, $g(n) \eqdef n + \ceil{\ln n}$, is another example.
\end{solution}


\ppart Give an example of $f,g$ such that
\[
f \neq O(g)\ \QAND\ g \neq O(f).
\]

\begin{solution}
Let $c_{\text{even}}$ be the characteristic function for the even numbers:
\[
c_{\text{even}}(n) = \rem{n,2}.
\]

Let
\[
f(n) \eqdef 1+ n\cdot c_{\text{even}}(n), \qquad g(n) \eqdef 1 + n\cdot c_{\text{even}}(n+1).
\]
which can also be described as
\[
f(n) \eqdef 1+ n\, \sin^2\paren{\frac{n\pi}{2}}, \qquad g(n) \eqdef 1+ n\, \cos^2\paren{\frac{n\pi}{2}}.
\]

\end{solution}

\eparts

\end{problem}
