\begin{problem}
Let $P$ be any set predicate, and define
\[
\mathcal{S}_P \eqdef  \set{S \in \text{Sets} \suchthat P(S)}.
\]

\bparts

\ppart Give an example $P$ such that $\mathcal{S}_P$ is not a set.

\begin{solution}
The simplest example would be when $P$ is the identically true
predicate \True, so $\mathcal{S}_{\True} =\ \text{Sets}$.
\end{solution}


\ppart Give an example $P$ such that $\mathcal{S}_P$ is a set.

\begin{solution}
The simplest example would be when $P$ is the identically false
predicate \False, so $\mathcal{S}_{\False} =\ \emptyset$.
\end{solution}

\ppart Prove that if $\mathcal{S}_P$ is a set, then $P(\mathcal{S}_P)$
must be false.

\begin{solution}
If $P(\mathcal{S}_P)$ was true, we would have $\mathcal{S}_P \in
\mathcal{S}_P$, which would be a violation of the Foundation axiom.

\end{solution}

\eparts

\end{problem}
