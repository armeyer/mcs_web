\begin{problem}\mbox{}

For each of the relations below, indicate whether it is
\emph{transitive} but not a partial order (\textbf{Tr}), a \emph{total
order} (\textbf{Tot}), a \emph{strict partial order} that is not total
(\textbf{S}), a \emph{weak partial order} that is not total (\textbf{W}),
or \emph{none} of the above (\textbf{N}).
\begin{itemize}

\item the ``is a subgraph of'' relation on graphs.  \hfill \examrule\\
(Note that every graph is considered a subgraph of itself.)
\end{itemize}


Let $f,g$ be nonnegative functions on the real numbers.
\begin{itemize}

\item  the ``Big Oh'' relation, $f=O(g)$, \hfill \examrule

\item  the ``Little Oh'' relation, $f=o(g)$, \hfill \examrule

\item the ``asymptotically equal'' relation, $f \sim g$. \hfill \examrule

%\set{(a,b) | a<b} \text{ on the set of integers }\quad & \brule{1in}\\
%% Added
%\set{(a,b) | a^2 \leq b^2} \text{ on the set of real numbers }\quad & \brule{1in}\\

\end{itemize}

\begin{solution}big Oh is \textbf{Tr}, little Oh is \textbf{S}, sim is
\textbf{Tr}, and subgraph is \textbf{W}.
\end{solution}

\end{problem}
