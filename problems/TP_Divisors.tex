\bparts

\ppart
What are all the divisors of 12?

\begin{solution}
There are twelve: $\pm 1$,  $\pm 2$,  $\pm 3$,  $\pm 4$,  $\pm 6$, and  $\pm 12$.
\end{solution}

\ppart
What are all the positive divisors of 12?

\begin{solution}
There are six: 1, 2, 3, 4, 6, and 12.
\end{solution}

\ppart
What are all the prime divisors of 12?

\begin{solution}
There are two: 2 and 3.
\end{solution}

\eparts

\end{problem}
