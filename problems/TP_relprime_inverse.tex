\begin{problem}
Prove that $k\in [0,n)$ has an inverse modulo $n$ iff it has an
  inverse in $\Zmod{n}$.

\begin{solution}

\begin{align*}
1 \equiv jk \pmod{n} & \qiff 1 = j \cdot_n k\\
 & \qiff 1 = \rem{j}{n} \cdot_n k.
\end{align*}
\end{solution}

\end{problem}
