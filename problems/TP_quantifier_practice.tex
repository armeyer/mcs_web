\begin{problem} 
Express each of the following statements using quantifiers, logical
connectives, and/or the following predicates
\begin{description}
\item[$P(x)$:] $x$ is a monkey,
\item[$Q(x)$:] $x$ is a 6.042 TA,
\item[$R(x)$:] $x$ comes from the 23rd century,
\item[$S(x)$:] $x$ likes to eat pizza,
\end{description}
where $x$ ranges over all living things.
\bigskip

\begin{problemparts}
\ppart\label{nomonkey} No monkeys like to eat pizza.

\begin{solution}
$\forall x.\ P(x) \QIMPLIES \QNOT(S(x))$
\end{solution}

\ppart\label{no23} Nobody from the 23rd century dislikes eating pizza.

\begin{solution}
$\forall x.\ R(x) \QIMPLIES S(x)$
\end{solution}

\ppart\label{allta} All 6.042 TAs are monkeys.

\begin{solution}
$\forall x.\ Q(x) \QIMPLIES P(x)$
\end{solution}

\ppart\label{nota}  No 6.042 TA comes from the 23rd century.

\begin{solution}
$\forall x.\ Q(x) \QIMPLIES \QNOT(R(x))$
\end{solution}

\ppart Does part~\eqref{nota} follow logically from
parts~\eqref{nomonkey},~\eqref{no23},~\eqref{allta}?  If so, give a
proof.  If not, give a counterexample.

\begin{solution}
\begin{center}
\begin{align*}
Q(x) & \QIMPLIES P(x), & \text{(from~\eqref{allta})} \\
P(x) & \QIMPLIES \QNOT(S(x)), & \text{(from~\eqref{nomonkey})} \\
\QNOT(S(x)) & \QIMPLIES \QNOT(R(x)), & \text{(contrapositive of~\eqref{no23})}\\ 
\text{therefore} &\ Q(x) \QIMPLIES \QNOT R(x).
\end{align*}
\end{center}
\end{solution}

\ppart Translate into English: $(\forall x) (R(x) \vee S(x) \implies
Q(x))$.

\begin{solution}
Anyone who is from the 23rd century or likes to eat pizza is a 6.042
TA.
\end{solution}

\ppart Translate into English:
\[
[\exists x.\, R(x) \QAND \QNOT(Q(x))]
\QIMPLIES \forall x.\,(P(x) \QIMPLIES S(x)).
\]

\begin{solution}
If there is a person from the 23rd century who is not a 6.042 TA then
all monkeys like to eat pizza
\end{solution}

\begin{staffnotes}
Some students forgot leading quantifiers in the first parts (especially
$\forall x$). On the last part, many students mistakenly said that
the entity mentioned in the first `if' would be a monkey that liked
pizza. Note that this is not the same as the correct statement.
\end{staffnotes}

\end{problemparts}
\end{problem}
