\begin{problem}
Find the last digit of $7^{7^{7^7}}$.

\begin{solution}
\textbf{1}.

The last digit of $7^{7^{7^7}}$ is $\rem{7^{7^{7^7}}}{10}$.  Because 7
and 10 are relatively prime,
\[
7^{7^{7^7}} \equiv 7^{\rem{7^{7^7}}{\phi(10)}} = 7^{\rem{7^{7^7}}{4}} \pmod{10}.
\]
So now we need to find $\rem{7^{7^7}}{4}$.  Since 7 and 4 are
relatively prime, 
\[
7^{7^7} \equiv 7^{\rem{7^7}{\phi(4)}} = 7^{\rem{7^7}{2}} \pmod{4}.
\]
But $\rem{7^7}{2}$ is clearly 1, so
\[
7^{7^7} \equiv 7^1 \equiv 3 \pmod{4},
\]
and therefore
\[
\rem{7^{7^7}}{4} = 3.
\]
Finally,
\[
7^{7^{7^7}} \equiv 7^{\rem{7^{7^7}}{4}} = 7^3 \equiv 1 \pmod{10},
\]
so $\rem{7^{7^{7^7}}}{10} = 1$.

\end{solution} 

\end{problem}
