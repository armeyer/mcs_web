\begin{problem}
The Fibonacci numbers $F(0), F(1), F(2), \dots$ are defined as follows:
\begin{align}
  F(0) & \eqdef  0, \notag\\
  F(1) & \eqdef  1, \notag\\
  F(n) & \eqdef  F(n - 1) + F(n - 2) & \text{for } n \geq 2.\label{fibndef}
\end{align}

  Thus, the first few Fibonacci numbers are 0, 1, 1, 2, 3, 5, 8, 13, and
  21.  Prove by induction that for all $n \geq 1$,
\begin{equation}\label{fn-1fn+1}
F(n-1) \cdot F(n+1) - F(n)^2 =  (-1)^n.
\end{equation}

\begin{solution}
\begin{proof}
  The proof is by induction on $n$ with equation~\eqref{fn-1fn+1} as
  induction hypothesis, $P(n)$.

\inductioncase{base case:} $n=1$.  $P(1)$ holds because
\[
  F(1-1) \cdot F(1+1) - F(1)^2 = 0 \cdot 1 - 1^2  =  -1^1.
\]

\inductioncase{inductive step:} We show that $P(n+1)$ is true, assuming that
$n \geq 1$ and $P(n)$ is true.  To do this, we consider
equation~\eqref{fn-1fn+1} with $n$ replaced by $n+1$ and then
transform the left hand side of this new equation into the right hand
side, namely, $(-1)^{n+1}$. So starting with $n+1$ for $n$ in left
side of~\eqref{fn-1fn+1}, we have
\begin{align*}
\lefteqn{F((n+1) - 1) \cdot F((n+1) + 1) - F(n+1)^2}\\
        & =  F(n) \cdot F(n+2) - F(n+1)^2 
             & \text{(simplify $(n+1)\pm 1$)}\\
        & =  F(n) \cdot (F(n+1) + F(n))\\
        & \qquad - F(n+1) \cdot (F(n) + F(n-1))
                & \text{(def.\ of $F(n+2)$ \& $F(n+1)$)}\\
        & =  F(n) \cdot F(n+1) + F(n) \cdot F(n)\\
        & \qquad -  F(n+1) \cdot F(n) - F(n+1) \cdot F(n-1)
                &\text{(distribute multipliers)}\\
        & =  F(n) \cdot F(n) - F(n+1) \cdot F(n-1) 
                & \text{(cancel $F(n) \cdot F(n+1$))}\\
        & =  - (-1)^n
                &\text{(induction hypothesis~\eqref{fn-1fn+1})}\\
        & =  (-1)^{n+1}.
\end{align*}
This proves $P(n+1)$, completing the inductive step.

We conclude by induction that equation~\eqref{fn-1fn+1} holds for all
$n \geq 1$.
\end{proof}
\end{solution}

\end{problem}
