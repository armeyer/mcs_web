\begin{problem}
  
\bparts
\ppart
Show that
\[
(an)^{b/n} \sim 1.
\]
where $a,b$ are positive constants and $\sim$ denotes asymptotic equality.
\hint $an=a2^{\log_2 n}$.

\examspace{4in}

\begin{solution}
  \[
  (an)^{b/n} = \paren{a^b}^{1/n} \cdot 2^{(b\log_2 n)/n} \to 1 \cdot 2^0 = 1,
  \]
  as $n \to \infty$.
\end{solution}

\ppart
Show that
\[
\sqrt[n]{n!} = \Theta(n).
\]

\examspace{4in}

\begin{solution}
  \begin{align*}
  \sqrt[n]{n!} & 
  \sim \paren{(2 \pi n)^{\frac{1}{2}} \paren{\frac{n}{e}}^n}^{1/n}
  & \text{(Stirling)}\\
  & \sim (2 \pi n)^{\frac{1}{2n}} \frac{n}{e}\\
  & \sim 1 \cdot  \frac{n}{e}  & \text{part (a)}\\
               & = \Theta(n)
  \end{align*}
\end{solution}

\eparts

\end{problem}
