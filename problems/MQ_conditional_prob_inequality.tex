\begin{problem}
A flip of Coin 1 is $x$ times as likely to come up Heads as a flip of
Coin 2.  A biased random choice of one of these coins made, where the
probability of choosing Coin 1 is $w$ times that of Coin 2.  The
random coin is flipped and comes up Heads.

\bparts

\ppart Restate the information above using probabilities and conditional
probabilities involving the events
\begin{align*}
  C1 & \eqdef \text{Coin 1 was the random choice},\\
  C2 & \eqdef \text{Coin 2 was the random choice},\\
  H  & \eqdef \text{the random coin come up Heads}.
\end{align*}

\examspace[1.0in]

\begin{solution}
We are given that
\begin{equation}\label{HC1x}
\prcond{H}{C1} = x \prcond{H}{C2},
\end{equation}
and
\[
\pr{C1} = w \pr{C2},
\]
\end{solution}

\ppart Prove that Coin 1 is more likely than Coin 2 to have been the
random coin iff
\[
wx \leq 1.
\]

\examspace[3.0in]

\begin{solution}
From~\eqref{HC1x} and the definition of conditional probability,
\[
\frac{\pr{H \QAND C1}}{\pr{C1}} = x \frac{\pr{H \QAND C2}}{\pr{C2}}.
\]
Substituting for $\pr{C1}$, we get
\[
\frac{\pr{H \QAND C1}}{w\pr{C2}} = x \frac{\pr{H \QAND  C2}}{\pr{C2}},
\]
which implies
\[
\pr{H \QAND C1} = wx\pr{H \QAND C2}.
\]
Dividing both sides of this inequality by $\pr{H}$, yields
\[
\prcond{C1}{H} = wx \prcond{C2}{H},
\]
so 
\[
\prcond{C1}{H} \geq \prcond{C2}{H} \QIFF wx \leq 1
\]

\end{solution}

\eparts

\end{problem}
