\begin{problem}
Let $\prec$ be a partial order on a set, $A$, and let
\[
A_k \eqdef \set{a \suchthat \dpth{a} =k}
\]
where $k \in \naturals$.

\bparts

\ppart Prove that $A_0,A_1,\dots$ is a parallel schedule for $\prec$
according to Definition~\bref{def:schedule}.

\begin{solution}

\begin{proof}
We have to show that if $a \in A_k$, $b \in A_j$ and $b \prec a$,
then $j < k$.

Now since $\dpth{b} = j$ by definition of $A_j$, there is a length $j$
chain, $C$, that ends at $b$.  Since $b$ is the maximum element of $C$
and $b \prec a$, the set $C \union \set{a}$ is a chain of length
$j+1$ that ends at $a$.  So $\dpth{a} \geq j+1$ by definition of depth, and
\[
k = \dpth{a} \geq j+1 > j.
\]
\end{proof}

\end{solution}

\ppart Prove that $A_k$ is an antichain.

\begin{solution}

\begin{proof}
Suppose $a \neq b$ are elements of $A$ and $\dpth{a} = \dpth{b}$.
We want to show that $a$ and $b$ are incomparable.

Suppose to the contrary that $a$ and $b$ are comparable, say, $b \prec
a$.  By definition of depth, there is a chain, $C$, of length
$\dpth{b}$ ending at $b$.  Since $b$ is the maximum element of $C$ and
$b \prec a$, the set $C \union \set{a}$ is a chain of length
$\dpth{b}+1$ that ends at $a$.  It follows that $\dpth{a} > \dpth{b}$,
contradicting the assumption that $\dpth{a} = \dpth{b}$.
\end{proof}

\end{solution}

\eparts
\end{problem}
