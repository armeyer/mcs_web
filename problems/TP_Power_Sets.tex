\begin{problem}

%% title: Power Sets

For any set $A$, let $\power{P}(A)$ be its \emph{power
set}, the set of all its subsets; note that $A$ is itself a member of
$\power{P}(A)$. Let $\emptyset$ denote the empty set.

\bparts

\ppart
The elements of $\power{P}(\set{1,2})$ are:

\begin{solution}
$\set{1,2}$, $\set{1}$, $\set{2}$, $\emptyset$
\end{solution}

\ppart
The elements of $\power{P}(\set{\emptyset, \set{\emptyset} })$ are:

\begin{solution}
$\set{\emptyset, \set{\emptyset} }$,
$\set{\emptyset}$,
$\{ \set{set\emptyset} }$,
$\emptyset$
\end{solution}

\ppart
How many elements are there in $\power{P}(\set{1, 2, \dots , 8})$?
\begin{solution}
$2^8$.
\end{solution}

\eparts

\end{problem}
