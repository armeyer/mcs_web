\begin{problem} Prove by induction:
\begin{equation}\label{1n2}
1+\frac{1}{4}+\frac{1}{9}+\cdots+\frac{1}{n^2}  <  2-\frac{1}{n},
\end{equation}
for all $n >1$.

\begin{solution}
\begin{proof}
(By Induction).  The induction hypothesis, $P(n)$, is the
inequality~\eqref{1n2}.

\inductioncaseax{Base Case} ($n = 2$):  The LHS of~\eqref{1n2} in this case is
$1 + 1/4$ and the RHS is $2 - 1/2$, and
\[
\text{LHS}\ = 5/4 < 6/4 = 3/2 =\ \text{RHS},
\]
so inequality~\eqref{1n2} holds, and $P(2)$ is proved.

\inductioncase{Inductive Step}:  Let $n \geq 2$ be a nonnegative integer, and
assume $P(n)$ in order to prove $P(n+1)$.  That is, we assume~\eqref{1n2}.
Adding $1/(n+1)^2$ to both sides of this inequality yields

\begin{align*}
\lefteqn{1+\frac{1}{4}+\cdots+\frac{1}{n^2}+\frac{1}{(n+1)^2}}\\
& < 2-\frac{1}{n} +\frac{1}{(n+1)^2}\\
& = 2 - \paren{\frac{1}{n} - \frac{1}{(n+1)^2}}\\
& = 2 - \paren{\frac{n^2 + 2n + 1-n}{n(n+1)^2}}\\
& = 2 - \frac{n^2 + n}{n(n+1)^2}-\frac{1}{n(n+1)^2}\\
& = 2 - \frac{1}{n+1} - \frac{1}{n(n+1)^2}\\
& < 2 - \frac{1}{n+1} & \text{(since $n > 0$)}.
\end{align*}
So we have proved $P(n+1)$.

This completes the induction case, and we conclude by Induction that
$\forall n \ge 2.\, P(n)$.

\end{proof}

Note that although the proof Inductive step only needs to work for $n
\geq 2$, it actually works for all $n > 0$.
\end{solution}
\end{problem}
