\begin{problem}
Assume the year has 365 days.  Answer the following questions with a
simple expression that may involve factorials, exponentials, and
binomial coefficients.

\bparts

\ppart How many possible pairs of birthdays are there?\hfill \examrule[0.7in]

\begin{solution}
\[
365^2
\]
\end{solution}


\ppart How many possible pairs of \emph{different} birthdays are there?\hfill \examrule[0.7in]

\begin{solution}
\[
365 \cdot 364
\]
\end{solution}

\ppart How many possible sets of two birthdays are there?\hfill \examrule[0.7in]

\begin{solution}
\[
\binom{362}{2}
\]
\end{solution}

\eparts

\end{problem}
