\begin{problem}
Let $R$ be a positive integer valued random variable such that
\[
\pdf_R(n)  = \frac{1}{cn^3},
\]
where
\[
c \eqdef \sum_{n=1}^{\infty} \frac{1}{n^3}.
\]

\bparts

\ppart\label{ppart:exf} Prove that $\expect{R}$ is finite.

\begin{solution}
	
\[
\expect{R} \eqdef \sum_{n \in \naturals^{+}} n \cdot \frac{1}{cn^3}
 = \sum_{n \in \naturals^{+}} \frac{1}{cn^2}
 < 1 + \int_{1}^{\infty} \frac{1}{cx^2}\,dx  =  1 + \frac{1}{c} < \infty.
\]

\end{solution}

\ppart\label{ppart:varinf}
Prove that $\variance{R}$ is infinite.

\begin{solution}
Since
\[
\variance{R} = \expect{R^2} - \expectsq{R},
\]
and $\expectsq{R} < \infty$ by part~\eqref{ppart:exf}, we need only show
that $\expect{R^2} = \infty$.  But
\[
\expect{R^2} \eqdef \sum_{n \in \naturals^{+}} n^2\frac{1}{cn^3}
 =  \sum_{n \in \naturals^{+}} \frac{1}{cn}
 = \frac{1}{c} \cdot \lim_{n \to \infty} H_n  = \infty.
\]
\end{solution}

A joking way to phrase the point of this example is ``The square root
of infinity may be finite.''  Namely, let $T \eqdef R^2$.  Then
\insolutions{the solution to} part~\eqref{ppart:varinf} implies that
$\expect{T} = \infty$ while $\expect{\sqrt{T}} < \infty$
by~\eqref{ppart:exf}.

\eparts

\end{problem}
