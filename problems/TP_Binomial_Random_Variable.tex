\begin{problem}

%% title: Binomial Random Variable

Let $B$ be the number of heads that come up on $2n$ independent tosses
of a fair coin.  \bparts

\ppart $\prob{B = n}$ is asymptotically equal to one of the
expressions given below.  Explain which one.
      
\begin{enumerate}

\item
$\frac{1}{\sqrt{2 \pi n}}$

\item
$\frac{2}{\sqrt{\pi n}}$

\item\label{f1spn-ans}
$\frac{1}{\sqrt{\pi n}}$
      
\item
$\sqrt{\frac{2}{\pi n}}$
      
\end{enumerate}

\begin{solution}

Choice~\ref{f1spn-ans}. %3.

We have that
\[
\prob{B = n} = \binom{2n}{n} \cdot 2^{-2n}.
\]

Using Stirling's Formula,
\[
\binom{2n}{n} \sim \frac{2^{2n H(1/2)}}{\sqrt{2 \pi (1/2) (1 - 1/2)
    2n}} = \frac{2^{2n}}{\sqrt{\pi n}}
\]

Thus, 
\[
\prob{B = n} \sim \frac{2^{2n}}{\sqrt{\pi n}} * 2^{-2n} =
\frac{1}{\sqrt{\pi n}}.
\]
\end{solution}

\eparts

\end{problem}
