\begin{problem}
\iffalse Here is a straightforward modification of the notion of independence. \fi

Events $A, B, C$ are said to be \emph{conditionally
  independent} if 
\[
\prcond{A \intersect B}{C} = \prcond{A}{C} \cdot \prcond{B}{C} .
\]

Give a counterexample disproving the following

\begin{falseclm*}
If $A$ and $B$ are conditionally independent given $C$, and are also conditionally independent
given $D$, then $A$ and $B$ are conditionally independent given $C \intersect D$.
\end{falseclm*}

\hint Choose $A,B,C,D$ 3-way but not 4-way independent.

\begin{solution}
Choose $A,B,C,D$ 3-way but not 4-way independent.  Then $A$ and $B$
are conditionally independent, given $C$ or given $D$ by 3-way independence.  But
\begin{align*}
\prcond{A \intersect B}{C \intersect D}
 & \eqdef \frac{\pr{ A \intersect B \intersect C\intersect D}}{\pr{C \intersect D}}\\
 & \neq \frac{\pr{A} \cdot \pr{B} \cdot \pr{C} \cdot \pr{D}}{\pr{C \intersect D}}
      & \text{(not 4-way independent)}\\
 & = \frac{\pr{A} \cdot \pr{B} \cdot \pr{C} \cdot \pr{D}}{\pr{C} \cdot \pr{D}}
      & \text{(2-way independence of $C,D$)}\\
 & = \pr{A} \cdot \pr{B}\\
 & = \prcond{A}{C \intersect D} \cdot \prcond{B}{C \intersect D}
      & \text{(3-way independence)},
\end{align*}
so $A$ and $B$ are not independent given $C \intersect D$.
\end{solution}


\end{problem}
