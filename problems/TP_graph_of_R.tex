\begin{problem}
For a binary relation, $R: A \to B$, some properties of $R$ can be
determined from just the arrows of $R$, that is, from $\graph{R}$, and
others require knowing if there are elements in the domain, $A$, or the
codomain, $B$, that don't show up in $\graph{R}$.  For each of the
following possible properties of $R$, indicate whether it is always
determined by
\begin{enumerate}
%\setcounter{enumi}{1}
\item\label{justg} $\graph{R}$ alone,
\item\label{atoo} $\graph{R}$ and $A$ alone,
\item\label{btoo} $\graph{R}$ and $B$ alone,
\item\label{all3} all three parts of $R$.
\end{enumerate}

Properties:

\bparts
\ppart surjective
\begin{solution}
\ref{btoo}
\end{solution}


\ppart injective

\begin{solution}
\ref{justg}
\end{solution}

\ppart total

\begin{solution}
\ref{atoo}
\end{solution}


\ppart function

\begin{solution}
\ref{justg}
\end{solution}

\ppart bijection
\begin{solution}
\ref{all3}
\end{solution}

\eparts

\end{problem}
