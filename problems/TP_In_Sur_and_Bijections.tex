\begin{problem}

%% type: short-answer
%% title: In- ,Sur-, and Bijections

For each of the following real-valued functions on the
real numbers, indicate whether it is a bijection, a surjection but not a bijection, an injection but not a
bijection, or neither an injection nor a surjection.

\bparts

\ppart
$x\rightarrow x+2$

\begin{solution}
Bijection.
\end{solution}

\ppart
$x\rightarrow 2x$

\begin{solution}
Bijection.
\end{solution}

\ppart
$x\rightarrow x^2$

\begin{solution}
Neither.

Not a surjection, since negative numbers are not squares of reals, and
not an injection since $(-1)^2=1^2$.
\end{solution}

\ppart
$x\rightarrow x^3$

\begin{solution}
Bijection.
\end{solution}

\ppart
$x\rightarrow \sin x$

\begin{solution}
Neither.

Not a surjection, since $-1 \leq \sin x \leq 1$, and not an injection,
since $\sin 0 = \sin 2\pi$.
\end{solution}

\ppart
$x\rightarrow x\sin x$

\begin{solution}
Surjection.

Not an injection, since $0 \sin 0 = 2 \pi \sin 2 \pi$.
\end{solution}

\ppart
$x\rightarrow e^x$

\begin{solution}
Injection.

Not a surjection, since $e^x$ is always positive for real values of
$x$.
\end{solution}

\eparts

\end{problem}
