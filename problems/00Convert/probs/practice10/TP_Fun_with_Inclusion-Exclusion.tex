\begin{problem}

%% type: short-answer
%% title: Fun with Inclusion-Exclusion

If a positive integer \emph{not} exceeding 100 is selected at random,
what is the probability that it is divisible by 5 or 7?

\begin{solution}

0.32

\begin{align*}
\pr{\text{divisible by 5 or by 7}}
    & = \pr{\text{divisible by 5}} + \pr{\text{divisible by 7}}
         - \pr{\text{divisible by 5 and 7}} \\
    & = 20/100 + 14/100 - 2/100 \\
    & = 32/100 = 0.32
\end{align*}

\end{solution}
