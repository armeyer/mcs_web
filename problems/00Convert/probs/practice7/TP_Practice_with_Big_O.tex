\begin{problem}

%% type: short-answer
%% title: Practice with Big O

Find the least natural number, $n$, such that $f(x)$ is $O(x^{n})$
when $f$ is defined by each of the expressions below.  If there is
none, enter ``none''.

\bparts

\ppart
$2x^{3} + (\log x) x^{2}$

\begin{solution}
3

$2x^{3}$ grows as fast as $x^{3}$

$(\log x) x^{2}$ grows strictly slower than $x^{3}$.

\end{solution}

\ppart

$2x^{2}$ + $(\log x) x^{3}$

\begin{solution}
4

$2x^{2}$ grows as fast as $x^{2}$

$(\log x) x^{3}$ grows strictly faster than $x^{3}$ but strictly
slower than $x^{4}$.

\end{solution}

\ppart
$(1.1)^{x}$

\begin{solution}
none

$(1.1)^{x}$ grows strictly faster than any polynomial

\end{solution}

\ppart
$(0.1)^{x}$

\begin{solution}

As $x$ goes to infinity, $(0.1)^{x}$ goes to~0.  So it grows strictly
slower than any constant (same as a polynomial of degree~0).

\end{solution}

\ppart
$(x^{4} + x^{2} + 1) / (x^{3} + 1)$

\begin{solution}
1

This fraction grows as fast as $x^{4}/x^{3}$.

\end{solution}

\ppart

$(x^{4} + 5 \log x) / (x^{4} + 1)$

\begin{solution}

This fraction grows as fast as $x^{4}/x^{4}$.

\end{solution}

\ppart
$2^{(3 \log_{2} x^{2})}$

\begin{solution}
6

\begin{align*}
2^{(3 \log_{2} x^{2})} 
    & = 2^{ (\log_{2} (x^{2})^{3})} \\
    & = 2^{ (\log_{2} (x^{2 \cdot 3}))} \\
    & = 2^{ (\log_{2}(x^{6}))}\\
    & = x^{6}
\end{align*}

\end{solution}

\eparts

\end{problem}
