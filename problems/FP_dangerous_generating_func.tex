\begin{problem}
Every day in the life of Dangerous Dan is a potential disaster:
%
\begin{itemize}

\item Dan may or may not spill his breakfast cereal on his computer
keyboard.

\item Dan may or may not fall down the front steps on his way out the
door.

\item Dan stubs his toe zero or more times.

\item Dan blurts something foolish an even number of times.

\end{itemize}
%
Let $T_n$ be the number of different combinations of $n$ mishaps Dan
can suffer in one day.  For example, $T_3 = 7$, because there are
seven possible combinations of three mishaps:
%
\[
\begin{array}{rccccccc}
\text{spills} & 0 & 1 & 0 & 1 & 1 & 0 & 0 \\
\text{falls}  & 0 & 0 & 1 & 1 & 0 & 1 & 0 \\
\text{stubs}  & 3 & 2 & 2 & 1 & 0 & 0 & 1 \\
\text{blurts} & 0 & 0 & 0 & 0 & 2 & 2 & 2
\end{array}
\]

\bparts

\iffalse
\ppart[5] Fill in the boxes below with the correct quantities.
%
\[
T_0 = \fbox{\rule{0.75in}{0in}\rule[-0.3in]{0in}{0.75in}} \qquad
T_1 = \fbox{\rule{0.75in}{0in}\rule[-0.3in]{0in}{0.75in}} \qquad
T_2 = \fbox{\rule{0.75in}{0in}\rule[-0.3in]{0in}{0.75in}}
\]

\examspace[1.5in]
\begin{solution}
$T_0 = 1$, $T_1 = 3$, $T_2 = 4$
\end{solution}
\fi

\ppart[8] Express the generating function 
\[
T(x) \eqdef T_0 + T_1x + T_2x^2 + \cdots
\]
as a quotient of polynomials.

\begin{center}
\exambox{2.0in}{1.0in}{-0.4in}
\end{center}

\examspace[3.0in]

\begin{solution}
We multiply the generating functions for
spills $(1 + x)$, falls $(1 + x)$, stubs $(1 + x + x^2 + \cdots = 1 /
(1 - x))$, and blurts $(1 + x^2 + x^4 + \cdots = 1 / (1 - x^2))$:

\[
\frac{(1+x)^2}{(1-x)(1-x^2)} = \frac{1+x}{(1-x)^2}
\]

\end{solution}

\ppart[5] Put integers in the boxes that make this equation true:
%
{\Large
\[
g(x) = \frac{\frame{\rule{0.75in}{0in}\rule{0in}{0.75in}}}{1-x}
     + \frac{\frame{\rule{0.75in}{0in}\rule{0in}{0.75in}}}{(1-x)^2}
\]}

\examspace[3in]

\begin{solution}
$-1, 2$
\end{solution}

\ppart[7]  Write a closed-form expression for $T_n$:

\exambox{2.0in}{1.0in}{-0.4in}
\examspace[0.5in]

\iffalse Remember that $1/(1-x)^2$ generates the sequence $\and{1, 2, 3,
\ldots}$.\fi

\begin{solution}
$2(n+1)-1 = 2n + 1$
\end{solution}

\eparts

\end{problem}
