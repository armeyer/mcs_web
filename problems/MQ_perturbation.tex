\begin{problem}

\begin{problemparts}

\problempart

Denote by $R$ the following geometric series, for which we know a closed-form 
expression. 
\begin{align*}
R & = \sum_{i=0}^n{z^i} \\
  & = 1 + z + z^2 + \ldots + z^n
\end{align*}
Find a closed-form expression for the sum
\begin{align*}
S & = \sum_{i=1}^n{i z^i}\\
  & = z + 2 z^2 + 3 z^3 + \ldots + n z^n
\end{align*}
Please leave your final answer in terms of $z$, $n$, and $R$.  Your answer must not
contain any derivatives of $R$.
\begin{solution}
\begin{align*}
z S & = z^2 + 2 z^3 + 3 z^4 + \ldots + n z^{n+1} \\
S - zS & = z + z^2 + z^3 + \ldots + z^n - n z^{n+1} \\
       & = R - 1 - n z^{n+1} \\
S & = \frac{R - 1 - n z^{n+1}}{1 - z}
\end{align*}
\end{solution}

\problempart

Let
\begin{align*}
R & = \sum_{i=0}^n{z^i} \\
  & = 1 + z + z^2 + \ldots + z^n
\end{align*}
and
\begin{align*}
T & = \sum_{i=1}^n{i z^i} \\
  & = z + 2 z^2 + 3 z^3 + \ldots + n z^n
\end{align*}
In class, we found closed-form expressions for both these series.

Find a closed-form expression for the sum
\begin{align*}
S & = \sum_{i=1}^n{i^2 z^i}\\
  & = z + 4 z^2 + 9 z^3 + \ldots + n^2 z^n
\end{align*}
Please leave your final answer in terms of $z$, $n$, $R$, and $T$.  Your answer must not
contain any derivatives of $R$ or $T$.
\begin{solution}
\begin{align*}
z S & = \sum_{i=1}^n{i^2 z^{i+1}} \\
S - zS & = \sum_{i=1}^n{i^2 z^i} - \sum_{i=1}^n{i^2 z^{i+1}} \\
       & = \paren{z + \sum_{i=2}^n{i^2 z^i}} - \paren{n^2 z^{n+1} + \sum_{i=2}^{n}{(i-1)^2 z^{i}}} \\
       & = z - n^2 z^{n+1} +  \sum_{i=2}^n{\paren{i^2-(i-1)^2} z^i} \\
       & = z - n^2 z^{n+1} +  \sum_{i=2}^n{\paren{i-(i-1)}\paren{i+(i-1)} z^i} \\
       & = z - n^2 z^{n+1} +  \sum_{i=2}^n{\paren{2i-1} z^i} \\
       & = z - n^2 z^{n+1} +  2\sum_{i=2}^n{i z^i} - \sum_{i=2}^n{z^i} \\
       & = z - n^2 z^{n+1} +  \paren{2\sum_{i=1}^n{i z^i} - 2 z} - \paren{\sum_{i=0}^n{z^i} - 1 - z} \\
       & = z - n^2 z^{n+1} +  \paren{2T - 2z} - \paren{R - 1 - z} \\
       & = - n^2 z^{n+1} + 2T - R + 1 \\
     S & = \frac{2T - R - n^2 z^{n+1} + 1}{1-z}
\end{align*}
\end{solution}

\problempart

Denote by $R(z)$ the following geometric series, for which we know a closed-form
expression.
\begin{align*}
R(z) & = \sum_{i=0}^n{z^i} \\
     & = 1 + z + z^2 + \ldots + z^n
\end{align*}
Find a closed-form expression for the sum
\begin{align*}
S(z) & = \sum_{i=1}^n{i z^{2i}}\\
     & = z^2 + 2 z^4 + 3 z^6 + \ldots + n z^{2n}
\end{align*}
Please leave your final answer in terms of $z$, $n$, and $R$.
\begin{solution}
\begin{align*}
z^2 S(z) & = z^4 + 2 z^6 + 3 z^8 + \ldots + n z^{2n+2} \\
S(z) - z^2 S(z) & = z^2 + z^4 + z^6 + \ldots + z^{2n} - n z^{2n+2} \\
             & = \paren{z^2} + \paren{z^2}^2 + \paren{z^2}^3 + \ldots + \paren{z^2}^n - n z^{2n+2} \\
             & = R\paren{z^2} - 1 - n z^{2n+2} \\
S & = \frac{R\paren{z^2} - 1 - n z^{2n+2}}{1 - z^2}
\end{align*}
\end{solution}
