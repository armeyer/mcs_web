\begin{problem}
Prove that for finite random walk graphs, the uniform distribution is
stationary iff the probabilities of the edges coming into each vertex
always sum to 1, namely
\begin{equation}\label{eq:sumuvin}
 \sum_{u \in \vin{v}} p(u,v)= 1,
\end{equation}
where $\vin{w} \eqdef \set{v \suchthat \diredge{v}{w} \text{ is an edge}}$.

\begin{solution}
For any distribution, $d$, let $\widehat{d}$ be the distribution after
one step.  So $d$ is stationary iff $d = \widehat{d}$.

Suppose $e$ is a uniform distribution, that is, $e(u) = e(v)$ for all
vertices $u,v$.  Then 
\begin{align*}
\widehat{e}(v)
 & \eqdef \sum_{u \in \vin{v}} p(u,v)e(u)\\
 & = \sum_{u \in \vin{v}} p(u,v)e(v)\\
 & = e(v)\sum_{u \in \vin{v}} e(u,v).
\end{align*}
Then $\widehat{e}(v) = e(v)$ iff~\eqref{eq:sumuvin} holds.
\end{solution}

\end{problem}
