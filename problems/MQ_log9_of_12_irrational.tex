\begin{problem} Prove that $\log_9 12$ is irrational.
%\hint Proof by contradiction.

\begin{solution}
\begin{proof}
  Suppose to the contrary that
  \[
  \textcolor{red}{\log_9 12 = \frac{m}{n}}
  \]
  for some integers $m, n$ where $n>0$.  So we have
\begin{align}
%  \log_9 12 & = m/n, \notag\\
  9^{\log_9 12} & = 9^{m/n}  & \text{(raising 9 to equal powers)},\notag\\
  12 & = 9^{m/n} & \text{(def of $\log_9$)},\notag\\
  12^n & = 9^m & \text{(raising both sides to the $n$th power)}.\label{12n9m}
\end{align}
But the left hand side of~\eqref{12n9m} is even since $n > 0$,
and the right hand side is not.

This contradiction implies that $\log_9 12$ must be irrational.
\end{proof}
\end{solution}
\end{problem}
