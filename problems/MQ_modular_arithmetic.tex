\begin{problem}

\bparts

\ppart
Prove that $22^{12001}$ has a multiplicative inverse modulo
175.

\examspace[2in]
\begin{solution}
The numbers 22 and 175 are relatively prime since $22= 2 \cdot 11$ and
$175 = 5^2 \cdot 7 $.  This could also be verified by computing
\[
\gcd(175, 22) = \gcd(22, \rem{175}{22}) = \gcd(22, 21) = \gcd(21, 1) = 1.
\]
Now unique factorization implies that
\[
\gcd(22^n, 175^m) = 1
\]
for any $m,n \in \naturals$ and so $22^n$ has a multiplicative inverse
modulo $175^m$.
\end{solution}

\ppart What is the value of $\phi(175)$, where $\phi$ is Euler's
function?

\examspace[2in]
\begin{solution}
Noting that $175=5^2\cdot 7$.  It follows
that $\phi(175)=(5^2 - 5^1)(7-1)= 20 \cdot 6 =120$.
\end{solution}

\ppart What is the remainder of $22^{12001}$ divided by 175?

\begin{solution}
Since 22 and 175 are relatively prime, we have by Euler's Theorem that
$22^{120} \equiv 1 \pmod {175}$, and so
\[
22^{12001} = \paren{22^{120}}^{100}\cdot 22 \equiv 1^{100}\cdot 22 \equiv 22 \pmod {175}.
\]
\end{solution}

\eparts
\end{problem}
