\begin{problem}

Prove by induction that every nonempty finite set of real numbers has
a minimum element.

\begin{staffnotes}
\hint Show that every set of $n>0$ real numbers has a minimum element.
\end{staffnotes}

\begin{solution}
\begin{proof}
A finite set by definition has $n$ elements for some $n \in \naturals$.
So the claim will follow if we prove that every set of $n \geq 1$ real
numbers has a minimum element.  That is, we need to prove that $\forall
n \geq 1.\, M(n)$, where
\begin{equation}\label{set_of_n_hypothesis}
M(n) \eqdef \text {every set of $n$ real numbers has a minimum element}.
\end{equation}

The proof will be by induction with induction hypothesis $M(n)$.

\inductioncase{Base case}: ($n=1$).  The minimum element of a set of
size one is that one element in the set.  This proves $M(1)$.

\inductioncase{Inductive step}: Let $R$ be a set of $n+1$ real numbers
for $n \geq 1$.  We will prove that $R$ has a minimum element.

Let $r$ be any element of $R$, and let $R'$ be $R' - \set{r}$.  Now
$R'$ has $n$ elements, and we may assume by induction that $R'$ has a
minimum element, $m$.  Then the minimum element of $R$ is simply the
smaller of $m$ and $r$.
\end{proof}

\end{solution}

\end{problem}
