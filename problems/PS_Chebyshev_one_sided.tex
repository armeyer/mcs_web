\begin{problem}
Prove the following ``one-sided'' version of \index{Chebyshev bound!one-sided}
the Chebyshev bound for deviation above the mean:
\begin{lemma*}[{One-sided Chebyshev bound}]
\[
\pr{R-\expect{R} \geq x}  \leq \frac{\variance{R}}{x^2+ \variance{R}}.
\]
\end{lemma*}

\hint Let $S_a \eqdef (R-\expect{R} + a)^2$, for $0 \leq a \in
\reals$.  So $R-\expect{R} \geq x$ implies $S_a \geq (x + a)^2$.
Apply \idx{Markov's bound} to $\pr{S_a \geq (x + a)^2}$.  Choose $a$
to minimize this last bound.

\iffalse
\[
\pr{(R-\expect{R} + a)^2 \geq (x + a)^2} \leq \frac{\expect{(R-\expect{R} + a)^2}}{(x+a)^2}.
\]\fi

\begin{solution}

\begin{proof}

\begin{align}
\pr{R-\expect{R} \geq x} & \leq \pr{S_a \geq (x + a)^2}
                          & \text{(from the hint)} \notag\\
   & \leq \frac{\expect{S_a}}{(x+a)^2}
                          & \text{(Markov bound)} \notag\\
   & =  \frac{\expect{(R-\expect{R})^2} +2a(\expect{R}-\expect{R}) + a^2}{(x+a)^2}
                & \text{(linearity of expectation)} \notag\\
   & =  \frac{\expect{(R-\expect{R})^2} + a^2}{(x+a)^2} \notag\\
   & =  \frac{\variance{R} + a^2}{(x+a)^2} & \text{(def of variance)\label{Va}}.
\end{align}

Setting the derivative of~\eqref{Va} with respect to $a$ to zero, we
find that $a = \variance{R}/x$ minimizes~\eqref{Va}.  Plugging in this
value for $a$, we have
\begin{align*}
\pr{R-\expect{R} \geq x}
    & \leq \frac{\variance{R} + \paren{\variance{R}/x}^2}{\paren{x+ \variance{R}/x}^2}
       & \text{($a = \variance{R}/x$ in~\eqref{Va})}\\
    & = \frac{\variance{R}x^2 + \varsq{R}}{\paren{x^2+ \variance{R}}^2}
       & \text{(multiply by $x^2/x^2$)}\\
    & = \frac{\variance{R}(x^2 + \variance{R})}{\paren{x^2+ \variance{R}}^2}\\
    & = \frac{\variance{R}}{x^2+ \variance{R}}.
\end{align*}

\end{proof}

\end{solution}

\end{problem}
