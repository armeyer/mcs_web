\begin{problem}
Show that there is an integer $x$ such that
\[
ax \equiv b \pmod n
\]
iff $\gcd(a,n) \divides b$.

\begin{solution}
The proof follows almost immediately from the fact that the linear
combinations of $c,d \in \integers$ are the same as the multiples of
$\gcd(c,d)$\inbook{, Corollary~\bref{cor:lin-comb}}.

\begin{align*}
\exists x.\ ax \equiv b \pmod n\quad 
  & \QIFF\ \exists x.\, \exists y. ax + ny = b\\
  & \QIFF\ \text{$b$ is a linear combination of $a$ and $n$}\\
  & \QIFF\ \text{$b$ is a multiple of $\gcd(a,n)$}\\
  & \QIFF\ \gcd(a,n) \divides b.
\end{align*}

\end{solution}
\end{problem}
