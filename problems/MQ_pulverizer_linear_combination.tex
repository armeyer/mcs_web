\begin{problem}

\bparts

\ppart Use the \idx{Pulverizer} to find $\gcd(84,108)$

\examspace[3in]

\begin{solution}
Here is the table produced by the Pulverizer:
\[
\begin{array}{ccccrcl}
x & \quad & y & \quad & \rem{x}{y} & = & x - q \cdot y \\ \hline
 108  &&  84  &&  24  & = &  1 \cdot 108 - 1 \cdot 84   \\
  84  &&  24  &&  12  & = & -3 \cdot 108 + 4 \cdot 84   \\
  24  &&  12  &&  0  & & 
\end{array}
\]
\end{solution}

\ppart Find integers $x$, $y$ with $0 \leq y < 84$ such that
\[
x \cdot 84 + y \cdot 108 = \gcd(84,108).
\]

\examspace[2in]

\begin{solution}
From the table above, 
\[
4 \cdot 84 - 3 \cdot 108 = \gcd(84,108).
\]
Therefore,
\[
(4 - 108 \cdot k) \cdot 84 + (-3 + 84 \cdot k) \cdot 108 = \gcd(84,108).
\]
So, letting $k = 1$, $(x,y) = (4 - 108 \cdot 1, -3 + 84 \cdot 1) = (-104, 81)$ works.
\end{solution}

\ppart Is there a multiplicative inverse of 84 in $\Zmod{108}$?  If
not briefly explain why, otherwise find it.

\begin{solution}
  There is no inverse of 84 modulo 108.  The inverse of $a$ modulo $m$ exists 
  iff $gcd(a,m) = 1$. Clearly $gcd(84, 108) = 12 \neq 1$, so there is no 
  inverse of 84 modulo 108.
\end{solution}

\eparts
\end{problem}
