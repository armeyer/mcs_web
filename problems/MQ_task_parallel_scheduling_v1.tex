\begin{problem}
Let $\{A, ..., H\}$ be a set of tasks that we must complete. The following DAG describes which tasks must be done before others, where there is an arrow from $a$ to $b$ iff $a$ must be done before $b$.

\begin{figure}[h]
\graphic[height=2.5in]{miniquiz5-p2}
\end{figure}

\bparts
\ppart
Write the longest chain.
\begin{solution}
A, D, E, G
\end{solution}
\examspace[3cm]

\ppart
Write the longest antichain.
\begin{solution}
A, B, C
\end{solution}
\examspace[3cm]

\ppart If we allow parallel scheduling, and each task takes 1 minute
to complete, what is the minimum amount of time needed to complete all
tasks?

\begin{solution}
This should be the length of the longest chain, 4 minutes.
\end{solution}

\eparts

\end{problem}
