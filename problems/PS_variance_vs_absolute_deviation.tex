\begin{problem}
\inhandout{The \term{expected absolute deviation} of a real-valued random
variable, $R$, with mean $\mu$, is defined to be
\[
\expect{\, \abs{R - \mu}\, }.
\]}
Prove that the \idx{expected absolute deviation}\inbook{ given in
  Definition~\ref{def:expabsdev}} is always less than or equal to the
standard deviation, $\sigma$.  (For simplicity, you may assume that
$R$ is defined on a finite sample space.)

\hint Suppose the sample space outcomes are $\omega_1, \omega_2,
\dots, \omega_n$, and let
\begin{align*}
\mathbf{p} & \eqdef (p_1,p_2,\dots,p_n) \quad \text{where } p_i =
              \sqrt{\prob{\omega_i}},\\
\mathbf{r} & \eqdef (r_1,r_2,\dots,r_n) \quad \ \text{where } r_i =
             \abs{R(\omega_i)-\mu}\sqrt{\prob{\omega_i}}.
\end{align*}
As usual, let $\mathbf{v} \cdot \mathbf{w} \eqdef \sum_{i=1}^n v_iu_i$
denote the \idx{dot product} of $n$-vectors $\mathbf{v}, \mathbf{w}$,
and let $\abs{\mathbf{v}}$ be the \idx{norm} of $\mathbf{v}$, namely,
$\sqrt{\mathbf{v} \cdot \mathbf{v}}$.

Then verify that
\[
\abs{\mathbf{p}}  = 1,\qquad \abs{\mathbf{r}}  = \sigma,\quad \text{ and }\quad
\expect{\, \abs{R - \mu}\, } = \mathbf{r} \cdot \mathbf{p}.
\]

\begin{solution}

\begin{equation}\label{=absp1}
\abs{\mathbf{p}}
    \eqdef \sqrt{\sum_{i=1}^n \paren{\sqrt{\prob{\omega_i}}}^2}
    = \sqrt{\sum_{i=1}^n \prob{\omega_i}}
    = \sqrt{1} = 1.
\end{equation}

\begin{align}
\abs{\mathbf{r}}
    & \eqdef \sqrt{\sum_{i=1}^n
        \brac{\abs{R(\omega_i)-\mu}\sqrt{\prob{\omega_i}}}^2}\notag\\
    & = \sqrt{\sum_{i=1}^n
        \abs{R(\omega_i)-\mu}^2\prob{\omega_i}}\notag\\
    & = \sqrt{\sum_{i=1}^n
        (R(\omega_i)-\mu)^2\prob{\omega_i}}\notag\\
    & = \sqrt{\expect{(R-\mu)^2}}\notag\\
    & = \sqrt{\variance{R}} = \sigma.\label{=absrsig}
\end{align}

\begin{align}
\mathbf{r} \cdot \mathbf{p}
    & \eqdef \sum_{i=1}^n
        \brac{\abs{R(\omega_i)-\mu}\sqrt{\prob{\omega_i}}}\sqrt{\prob{\omega_i}}\notag\\
    & = \sum_{i=1}^n \abs{R(\omega_i)-\mu}\prob{\omega_i}\notag\\
    & = \expect{\, \abs{R - \mu}\, }.\label{=rdotp}
\end{align}

Now, we have
\[
\expect{\, \abs{R - \mu}\, }
   = \mathbf{r} \cdot \mathbf{p}
   \leq \abs{\mathbf{r}} \abs{\mathbf{p}}
   = \sigma \cdot 1 = \sigma,
\]
where the first equality comes from~\eqref{=rdotp}, followed by
Cauchy-Schwarz inequality, with the second equality following
from~\eqref{=absrsig} and~\eqref{=absp1}.

\end{solution}
\end{problem}
